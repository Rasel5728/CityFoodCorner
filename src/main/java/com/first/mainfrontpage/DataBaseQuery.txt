CREATE DATABASE IF NOT EXISTS inventory_db;

USE inventory_db;

CREATE TABLE IF NOT EXISTS products (
    product_id   VARCHAR(50)  PRIMARY KEY,
    product_name VARCHAR(100) NOT NULL,
    type         VARCHAR(50),
    stock        INT          DEFAULT 0,
    price        DOUBLE       DEFAULT 0.0,
    status       VARCHAR(50),
    date         VARCHAR(50)
);
USE inventory_db;
SHOW TABLES;
